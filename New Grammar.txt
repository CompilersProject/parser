<PROGRAM> ::= <DEFINITIONS>
--------------------------------------------------------
<DEFINITIONS ::= <DEF1>
<DEF1> ::= <DEF> <DEFINITIONS1>
<DEFINITIONS1> ::= <DEFINITIONS>
				| EPSILON
--------------------------------------------------------				
<DEF> ::= <IDENTIFIER> ( <FORMALS> ) : <TYPE> <BODY>
--------------------------------------------------------
<FORMALS> ::= EPSILON
           | <NONEMPTYFORMALS>
--------------------------------------------------------						
 <NONEMPTYFORMALS> ::= <FORMAL> <NONEMPTYFORMALS1>
 <NONEMPTYFORMALS1> ::= , <NONEMPTYFORMALS>
					| EPSILON
--------------------------------------------------------					
<FORMAL> ::= <IDENTIFIER> : <TYPE>
--------------------------------------------------------
<BODY> ::= <PRINT> <BODY1>
		| <EXPR>
<BODY1> ::= <BODY>
--------------------------------------------------------
<TYPE> ::= integer
        | boolean
--------------------------------------------------------
<EXPR> ::= <EXPR1> | <EXPR2> | <SIMPLE-EXPR>
<EXPR1> ::= <SIMPLE-EXPR> > <EXPR>		
<EXPR2> ::= <SIMPLE-EXPR> = <EXPR>
< > > ::= not( < or = )
--------------------------------------------------------
<SIMPLE-EXPR> ::= <SIMPLE-EXPR1> | <SIMPLE-EXPR2> | <SIMPLE-EXPR3> | <SIMPLE-EXPR4> | <TERM>
<SIMPLE-EXPR1> ::= <TERM> or <SIMPLE-EXPR>
<SIMPLE-EXPR2> ::= <TERM> + <SIMPLE-EXPR>
<SIMPLE-EXPR3> ::= -<TERM> + <SIMPLE-EXPR>
<SIMPLE-EXPR4> ::= -<TERM>
--------------------------------------------------------
<TERM> ::= <TERM1> | <TERM2> | <TERM3> | <FACTOR>
<TERM1> ::= <FACTOR> and <TERM>
<TERM2> ::= <FACTOR> * <TERM>
<TERM3> ::= <FACTOR> * (1/<TERM>)
--------------------------------------------------------
<FACTOR> ::= if <EXPR> then <EXPR> else <EXPR> endif
			| not <EXPR>
			| <IDENTIFIER1>
			| <IDENTIFIER>
			| <LITERAL>
			
<IDENTIFIER1> ::= <IDENTIFIER> ( <ACTUALS> )
--------------------------------------------------------
<ACTUALS> ::= EPSILON
           | <NONEMPTYACTUALS>
--------------------------------------------------------
<NONEMPTYACTUALS> ::= <EXPR> , <NONEMPTYACTUALS1>
<NONEMPTYACTUALS1> ::= <NONEMPTYACTUALS>
					| EPSILON
--------------------------------------------------------			
<LITERAL> ::= <NUMBER>
           | <BOOLEAN>
--------------------------------------------------------
<PRINT> ::= print ( <EXPR> )
